#include <cstdio>
#include <iostream>

using namespace std;

typedef struct node{
    int data;
    struct node *next=NULL,*prev=NULL;
}node;

inline void display(node *head){
    node *cur=head;
    printf("Printing details: ");
    while (cur!=NULL) {
        printf("%d ",cur->data);
        cur=cur->next;
    }
}

inline void displayrev(node *end){
    node *cur=end;
    printf("\nPrinting details in reverse order: ");
    while (cur!=NULL) {
        printf("%d ",cur->data);
        cur=cur->prev;
    }
}

}

inline void deletionofnode(node *head){
    node *cur=head,*curnxt,*curprev;
    int temp=0;
    printf("Enter the value you want to delete: ");
    scanf("%d",&temp);
    while(cur!=nullptr){
        if(cur->data==temp){
            curprev->next=curnxt;
            curnxt->prev=curprev;
            free(cur);
            break;
        }
        cur=cur->next;
        curnxt=cur->next;
        curprev=cur->prev;
    }
}
