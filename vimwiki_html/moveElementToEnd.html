In this program we will be given a key and our main motive is to move all the elements which are equal to key at the end position so code accordingly and soft the problem. T.C. O(N) S.C. O(1) 

#include <iostream>

using namespace std;

inline void mymethod(int arr[],int size,int key){
    int start=0,end=size-1;
    while(start<end){
        if(arr[start]==arr[end]==key){
            end--;
        }
        else if(arr[start]==key){
            swap(arr[start],arr[end]);
            if(start!=0)
                start++;
            end--;
        }
        else if(arr[end]==key){
            swap(arr[start],arr[end]);
            end--;
            start++;
        }
        else{
            start++;
        }
    }
    for(int i=0;i<size;i++){
        cout<<i<<" ";
    }
}

inline void expertmethod(int arr[],int size,int key){
    int start=0,end=size-1;
    while(start<end){
        while(start<end && arr[end]==key){
            end--;
        }
        if(arr[start]==2){
            swap(arr[start],arr[end]);
        }
        start++;
    }
    for(int i=0;i<size;i++){
        cout<<arr[i]<<" ";
    }
}
